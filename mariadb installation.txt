use bellow link for create yum local repo using redaht dvd.

1[root@techbrown ~]# cd /

2[root@techbrown /]# mkdir localrepo

3 vim /etc/yum.repos.d/local.repo

copy below txt to repo file 

[rhel7]
name=rhel7
baseurl=file:///localrepo/
enabled=1
gpgcheck=0

4 createrepo /localrepo/

5 yum clean all
